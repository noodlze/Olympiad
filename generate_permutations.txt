//permuations of n! 
#define n  4
bool mark[n]= {0};
vector< vector<int> > permutation;
vector<int> temp;
void thu(int i) {
    for(int j=0; j< n; j++) {
        if(!mark[j]) {
            mark[j]=1;
            temp.push_back(j);
            if(i==3)
                permutation.push_back(temp);
            else
                thu(i+1);
            temp.pop_back();
            mark[j]=0;
            }
        }
    }
